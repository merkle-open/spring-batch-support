5   package com.namics.oss.spring.support.batch.writer;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   import org.springframework.batch.item.ItemWriter;
10  import org.springframework.data.repository.CrudRepository;
11  
12  import java.util.List;
13  
14  import static org.springframework.util.Assert.notNull;
15  
16  /**
17   * Writer to persist items using a spring data {@link org.springframework.data.repository.CrudRepository}.
18   *
19   * @author aschaefer
20   * @since 06.02.14 15:31
21   */
22  public class SpringDataItemWriter<T> implements ItemWriter<T> {
23  
24  	private static final Logger LOG = LoggerFactory.getLogger(SpringDataItemWriter.class);
25  
26  	protected CrudRepository<T, ?> repository;
27  
28  
29  	/**
30  	 * Writer to persist items using a spring data {@link org.springframework.data.repository.CrudRepository}.
31  	 *
32  	 * @param repository repository to use for persistence, must not be null
33  	 */
34  	public SpringDataItemWriter(CrudRepository<T, ?> repository) {
35  		notNull(repository, "spring data repository required.");
36  		this.repository = repository;
37  	}
38  
39  	/**
40  	 * Persist items
41  	 *
42  	 * @param items items to be persisted.
43  	 */
44  	@Override
45  	public void write(List<? extends T> items) {
46  		LOG.info("Write {} {} ", items.size(), items.iterator().hasNext() ? items.iterator().next().getClass() : "");
47  		if (LOG.isTraceEnabled()) {
48  			for (T item : items) {
49  				LOG.trace("write item {}", item);
50  			}
51  		}
52  		this.repository.saveAll(items);
53  	}
54  
55  }
