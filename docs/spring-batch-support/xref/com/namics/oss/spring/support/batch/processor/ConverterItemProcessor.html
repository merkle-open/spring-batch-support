5   package com.namics.oss.spring.support.batch.processor;
6   
7   import com.namics.oss.spring.convert.Converter;
8   import org.springframework.batch.item.ItemProcessor;
9   
10  import static org.springframework.util.Assert.notNull;
11  
12  /**
13   * Processor that uses a converter object to convert the item from one type to another.
14   *
15   * @param <Source> Source type to be converted to a target object
16   * @param <Target> Target type of result object as result of processing step
17   * @author aschaefer, Namics AG
18   * @see Converter
19   * @since 27.02.14 16:13
20   */
21  public class ConverterItemProcessor<Source, Target> implements ItemProcessor<Source, Target> {
22  
23  	protected Converter<Source, Target> converter;
24  
25  	/**
26  	 * Create a processor using a converter.
27  	 *
28  	 * @param converter Converter to use to process the item, must not be null.
29  	 */
30  	public ConverterItemProcessor(Converter<Source, Target> converter) {
31  		notNull(converter, "converter is required.");
32  		this.converter = converter;
33  	}
34  
35  	/**
36  	 * Transform item into target type using a {@link Converter} object to convert the item from one type to another.
37  	 *
38  	 * @param item to be processed
39  	 * @return transformed object
40  	 * @throws Exception on error
41  	 */
42  	@Override
43  	public Target process(Source item) throws Exception {
44  		return converter.convert(item);
45  	}
46  }
