5   package com.namics.oss.spring.support.batch.converter;
6   
7   import com.namics.oss.spring.convert.CollectionConverter;
8   import com.namics.oss.spring.support.batch.model.SimpleStep;
9   import org.springframework.batch.core.StepExecution;
10  
11  /**
12   * StepExecutionToSimpleStepConverter.
13   *
14   * @author lboesch, Namics AG
15   * @since 27.08.2014
16   */
17  public class StepExecutionToSimpleStepConverter extends CollectionConverter<StepExecution, SimpleStep> {
18  	@Override
19  	public SimpleStep convert(StepExecution stepExecution) {
20  		return convert(stepExecution, null);
21  	}
22  
23  	@Override
24  	public SimpleStep convert(StepExecution source, SimpleStep simpleStep) {
25  		SimpleStep target = simpleStep != null ? simpleStep : new SimpleStep();
26  		target.stepName(source.getStepName())
27  		      .status(source.getStatus().toString())
28  		      .readCount(source.getReadCount())
29  		      .writeCount(source.getWriteCount())
30  		      .commitCount(source.getCommitCount())
31  		      .rollbackCount(source.getRollbackCount())
32  		      .readSkipCount(source.getReadSkipCount())
33  		      .processSkipCount(source.getProcessSkipCount())
34  		      .writeSkipCount(source.getWriteSkipCount())
35  		      .startTime(source.getStartTime())
36  		      .endTime(source.getEndTime())
37  		      .exitStatus(source.getExitStatus().toString())
38  		      .lastUpdated(source.getLastUpdated())
39  		      .exitStatus(source.getExitStatus().toString());
40  		return target;
41  	}
42  }
