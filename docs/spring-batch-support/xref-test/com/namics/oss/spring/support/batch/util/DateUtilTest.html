28  		mutableDateTime.addHours(2);
29  		mutableDateTime.addMinutes(3);
30  		mutableDateTime.addSeconds(4);
31  		Date then = mutableDateTime.toDate();
32  		assertEquals("26 h : 3 m : 4.000 s", DateUtil.getDuration(now, then));
33  	}
